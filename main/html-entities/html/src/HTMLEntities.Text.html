import qualified Data.Text as Text
import HTMLEntities.Prelude

-- HTML-encodes the given char.
char :: Char -> Text
char :: Char -> Text
char =
  \case
    Char
'<' -> Text
"&lt;"
    Char
'>' -> Text
"&gt;"
    Char
'&' -> Text
"&amp;"
    Char
'"' -> Text
"&quot;"
    Char
'\'' -> Text
"&#39;"
    Char
x -> Char -> Text
Text.singleton Char
x
